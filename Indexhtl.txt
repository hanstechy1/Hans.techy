<script>

// 🔥 WEKA CONFIG YAKO
const firebaseConfig = {
  apiKey: "YOUR_KEY",
  authDomain: "YOUR_APP.firebaseapp.com",
  projectId: "YOUR_ID",
  storageBucket: "YOUR_APP.appspot.com",
};

firebase.initializeApp(firebaseConfig);

const auth = firebase.auth();
const db = firebase.firestore();
const storage = firebase.storage();

function show(page){
  document.querySelectorAll(".page").forEach(p=>p.classList.remove("active"));
  document.getElementById(page).classList.add("active");
}

// AUTH
function register(){
  auth.createUserWithEmailAndPassword(email.value,password.value)
  .then(()=>alert("Registered"));
}

function login(){
  auth.signInWithEmailAndPassword(email.value,password.value)
  .then(()=>alert("Logged in"));
}

// ADD BOOK
function addBook(){
  let file=bfile.files[0];
  let img=bimage.files[0];

  let ref1=storage.ref("books/"+file.name);
  let ref2=storage.ref("covers/"+img.name);

  ref1.put(file).then(()=>{
    ref1.getDownloadURL().then(fileURL=>{
      ref2.put(img).then(()=>{
        ref2.getDownloadURL().then(imgURL=>{
          db.collection("books").add({
            name:bname.value,
            price:bprice.value,
            file:fileURL,
            image:imgURL
          });
        });
      });
    });
  });
}

// LOAD BOOKS
function loadBooks(){
  db.collection("books").onSnapshot(snap=>{
    let html="";
    booksCount.innerText=snap.size;

    snap.forEach(doc=>{
      let b=doc.data();

      html+=`
      <div class="card">
      <img src="${b.image}">
      <h3>${b.name}</h3>
      <p>${b.price}</p>
      <button class="btn" onclick="addCart('${b.name}','${b.price}')">
      Buy
      </button>
      </div>`;
    });

    books.innerHTML=html;
  });
}

// CART
function addCart(name,price){
  let cart=JSON.parse(localStorage.getItem("cart"))||[];
  cart.push({name,price});
  localStorage.setItem("cart",JSON.stringify(cart));
}

function checkout(){
  alert("Proceed to payment");
}

// START
loadBooks();
